WITH base AS (
  SELECT reception_date, chart_id, a.grp_in_chart_id,
    final_staff, total_payment, arp_base_amount,
    nationality_type, category, visit_channel_1,
    IF(
      total_payment BETWEEN 500000
        AND IF(arp_base_amount <= 8000000, 800000, IFNULL(arp_base_amount,0)/10)
      AND coordinator_name IS NOT NULL AND consultation_blocked = FALSE
      AND (
        (treatment_category_1 IN ('제로네이트','마스필름') AND SAFE_CAST(unit_count_1 AS INT64) > 1)
        OR (treatment_category_2 IN ('제로네이트','마스필름') AND SAFE_CAST(unit_count_2 AS INT64) > 1)
        OR (treatment_category_3 IN ('제로네이트','마스필름') AND SAFE_CAST(unit_count_3 AS INT64) > 1)
      )
      AND is_consented = TRUE AND is_excluded_from_arp = FALSE
      AND is_refund_listed = FALSE
      AND SUM(sales_attribution_amount) OVER (PARTITION BY grp_in_chart_id) >= 500000
    , TRUE, FALSE) AS is_deposit
  FROM `tugether-data-warehouse-01.01_warehouse.00_dentweb_consult_merge` a
  WHERE reception_date >= '2026-03-01'
),
base_a1 AS (
  SELECT *,
    FIRST_VALUE(reception_date) OVER (
      PARTITION BY chart_id, grp_in_chart_id ORDER BY reception_date
    ) AS deposit_date
  FROM base WHERE is_deposit = TRUE
),
weekly_deposit AS (
  SELECT a.chart_id, a.grp_in_chart_id,
    DATE_TRUNC(MIN(b.deposit_date), WEEK(MONDAY))                             AS week_start,
    DATE_ADD(DATE_TRUNC(MIN(b.deposit_date), WEEK(MONDAY)), INTERVAL 6 DAY)  AS week_end,
    MIN(b.deposit_date) AS deposit_date, MAX(a.arp_base_amount) AS arp_base_amount,
    SUM(CASE WHEN DATE_TRUNC(b.deposit_date,WEEK(MONDAY))=DATE_TRUNC(a.reception_date,WEEK(MONDAY))
        THEN a.total_payment ELSE 0 END) AS payment_sum_in_week,
    SUM(CASE WHEN DATE_TRUNC(b.deposit_date,WEEK(MONDAY))=DATE_TRUNC(a.reception_date,WEEK(MONDAY))
        THEN a.total_payment ELSE 0 END)
    BETWEEN 500000 AND IF(MAX(a.arp_base_amount) <= 8000000, 800000, IFNULL(MAX(a.arp_base_amount),0)/10)
    AS is_deposit_in_week
  FROM `tugether-data-warehouse-01.01_warehouse.00_dentweb_consult_merge` a
  JOIN (SELECT chart_id, grp_in_chart_id, deposit_date FROM base_a1) b USING(chart_id, grp_in_chart_id)
  WHERE a.reception_date >= '2026-03-01'
  GROUP BY a.chart_id, a.grp_in_chart_id
)
SELECT week_start, week_end,
  COUNT(grp_in_chart_id)                                AS cnt_total,
  SUM(arp_base_amount)                                  AS arp_sum_total,
  SUM(payment_sum_in_week)                              AS payment_sum_total,
  COUNTIF(is_deposit_in_week)                           AS cnt_deposit,
  SUM(IF(is_deposit_in_week, arp_base_amount,       0)) AS arp_sum_deposit,
  SUM(IF(is_deposit_in_week, payment_sum_in_week,   0)) AS payment_sum_deposit
FROM weekly_deposit
WHERE week_start < '2026-06-08'
GROUP BY week_start, week_end
ORDER BY week_start

SELECT deposit_date,

  SUM(arp_base_amount)                                  AS arp_sum,
  SUM(payment_sum_in_week)                              AS payment_sum,
  COUNTIF(is_deposit_in_week)                           AS cnt_deposit,
  SUM(IF(is_deposit_in_week, arp_base_amount,       0)) AS arp_sum_deposit,
  SUM(IF(is_deposit_in_week, payment_sum_in_week,   0)) AS payment_sum_deposit
FROM weekly_deposit
WHERE week_start < '2026-06-08'
GROUP BY deposit_date
ORDER BY deposit_date
