WITH base AS (
  SELECT reception_date, chart_id, a.grp_in_chart_id, total_payment, arp_base_amount,
    IF(
      total_payment BETWEEN 500000
        AND IF(arp_base_amount <= 8000000, 800000, IFNULL(arp_base_amount,0)/10)
      AND coordinator_name IS NOT NULL AND consultation_blocked = FALSE
      AND (
        (treatment_category_1 IN ('제로네이트','마스필름') AND SAFE_CAST(unit_count_1 AS INT64) > 1)
        OR (treatment_category_2 IN ('제로네이트','마스필름') AND SAFE_CAST(unit_count_2 AS INT64) > 1)
        OR (treatment_category_3 IN ('제로네이트','마스필름') AND SAFE_CAST(unit_count_3 AS INT64) > 1)
      )
      AND is_consented = TRUE AND is_excluded_from_arp = FALSE AND is_refund_listed = FALSE
      AND SUM(sales_attribution_amount) OVER (PARTITION BY grp_in_chart_id) >= 500000
    , TRUE, FALSE) AS is_deposit
  FROM `tugether-data-warehouse-01.01_warehouse.00_dentweb_consult_merge` a
  WHERE reception_date >= '2026-03-01'   -- 일자별: '2026-05-11'
),
base_a1 AS (
  SELECT chart_id, grp_in_chart_id,
    MIN(reception_date) AS deposit_date,
    MAX(arp_base_amount) AS arp_base_amount
  FROM base WHERE is_deposit = TRUE
  GROUP BY chart_id, grp_in_chart_id
),
episode_summary AS (
  SELECT
    d.chart_id, d.grp_in_chart_id, d.deposit_date, d.arp_base_amount,
    DATE_TRUNC(d.deposit_date, WEEK(MONDAY))                             AS week_start,
    DATE_ADD(DATE_TRUNC(d.deposit_date, WEEK(MONDAY)), INTERVAL 6 DAY)  AS week_end,
    SUM(CASE WHEN DATE_TRUNC(d.deposit_date,WEEK(MONDAY))=DATE_TRUNC(a.reception_date,WEEK(MONDAY))
        THEN a.total_payment ELSE 0 END)                                 AS payment_in_week,
    SUM(CASE WHEN a.reception_date >= d.deposit_date AND a.reception_date <= '2026-06-23'
        THEN a.total_payment ELSE 0 END)                                 AS payment_cumul
  FROM `tugether-data-warehouse-01.01_warehouse.00_dentweb_consult_merge` a
  JOIN base_a1 d USING(chart_id, grp_in_chart_id)
  WHERE a.reception_date >= '2026-03-01'
  GROUP BY d.chart_id, d.grp_in_chart_id, d.deposit_date, d.arp_base_amount
)
-- 주차별: GROUP BY week_start, week_end  WHERE week_start < '2026-06-22'
-- 일자별: GROUP BY deposit_date          WHERE deposit_date <= '2026-06-23'
SELECT week_start, week_end,
  COUNT(grp_in_chart_id)                                                              AS cnt_total,
  SUM(arp_base_amount)                                                                AS arp_sum,
  COUNTIF(payment_in_week BETWEEN 500000 AND IF(arp_base_amount<=8000000,800000,IFNULL(arp_base_amount,0)/10)) AS cnt_dep_week,
  COUNTIF(payment_cumul   BETWEEN 500000 AND IF(arp_base_amount<=8000000,800000,IFNULL(arp_base_amount,0)/10)) AS cnt_dep_cumul,
  SUM(payment_in_week)                                                                AS payment_week_total,
  SUM(payment_cumul)                                                                  AS payment_cumul_total
FROM episode_summary
WHERE week_start < '2026-06-22'
GROUP BY week_start, week_end ORDER BY week_start
